#include <stdio.h>
#include <stdlib.h>
int compare(const void *a, const void *b) {
    return (*(int *)a - *(int *)b);
}
int main() {
    int n, k;
    scanf("%d %d", &n, &k);
    int *houses = (int *)malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) {
        scanf("%d", &houses[i]);
    }
    qsort(houses, n, sizeof(int), compare);
    int transmitters = 0;
    int i = 0;
    while (i < n) {
        int loc = houses[i] + k;
        while (i < n && houses[i] <= loc) {
            i++;
        }
        int transmitterPos = houses[i - 1];
        loc = transmitterPos + k;
        while (i < n && houses[i] <= loc) {
            i++;
        }
        transmitters++;
    }
    printf("%d\n", transmitters);
    free(houses);
    return 0;
}
