#include <stdio.h>
#include <stdlib.h>
typedef struct {
    int cost;
    int idx;
} Entry;
int main(void) {
    int t;
    if (scanf("%d", &t) != 1) return 0;
    while (t--) {
        int m, n;
        scanf("%d %d", &m, &n);
        Entry *seen = malloc(n * sizeof(Entry));
        int seenCount = 0;
        int found1 = -1, found2 = -1;
        for (int i = 0; i < n; i++) {
            int c;
            scanf("%d", &c);
            for (int j = 0; j < seenCount; j++) {
                if (seen[j].cost == m - c) {
                    found1 = seen[j].idx;
                    found2 = i + 1;
                    break;
                }
            }
            if (found1 != -1) {
                for (int k = i + 1; k < n; k++) scanf("%*d");
                break;
            }
            seen[seenCount].cost = c;
            seen[seenCount].idx = i + 1;
            seenCount++;
        }
        if (found1 > found2) {
            int tmp = found1;
            found1 = found2;
            found2 = tmp;
        }
        printf("%d %d\n", found1, found2);

        free(seen);
    }

    return 0;
}
