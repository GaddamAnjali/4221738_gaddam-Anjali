#include <stdio.h>
#include <stdlib.h>
int main(void) {
    int n;
    if (scanf("%d", &n) != 1) return 0;
    int *lead = malloc(n * sizeof(int));
    int distinct = 0;
    for (int i = 0; i < n; i++) {
        int score;
        scanf("%d", &score);
        if (i == 0 || score != lead[distinct - 1]) {
            lead[distinct++] = score;
        }
    }
    int m;
    scanf("%d", &m);
    int *player = malloc(m * sizeof(int));
    for (int i = 0; i < m; i++) {
        scanf("%d", &player[i]);
    }
    int idx = distinct - 1;
    for (int i = 0; i < m; i++) {
        int sc = player[i];
        while (idx >= 0 && sc >= lead[idx]) {
            idx--;
        }
        printf("%d\n", idx + 2);
    }
    free(lead);
    free(player);
    return 0;
}
