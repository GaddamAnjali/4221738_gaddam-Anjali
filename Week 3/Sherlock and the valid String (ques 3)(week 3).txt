#include <stdio.h>
#include <string.h>
#define YES "YES\n"
#define NO  "NO\n"
int main(void) {
    char s[100001];
    if (scanf("%100000s", s) != 1) return 0;
    int freq[26] = {0};
    for (int i = 0; s[i]; i++) {
        freq[s[i] - 'a']++;
    }
    int freqCount[100001] = {0};
    for (int i = 0; i < 26; i++) {
        if (freq[i] > 0) {
            freqCount[freq[i]]++;
        }
    }
    int uniqueFreqs = 0;
    int f1 = 0, f2 = 0, c1 = 0, c2 = 0;
    for (int f = 1; f <= 100000; f++) {
        if (freqCount[f] > 0) {
            if (uniqueFreqs == 0) {
                f1 = f;
                c1 = freqCount[f];
            } else if (uniqueFreqs == 1) {
                f2 = f;
                c2 = freqCount[f];
            }
            uniqueFreqs++;
        }
    }
    if (uniqueFreqs == 1) {
        printf(YES);
    } else if (uniqueFreqs == 2) {
        if ((f1 == 1 && c1 == 1) || (f2 == 1 && c2 == 1)) {
            printf(YES);
        }
        else if ((f1 - f2 == 1 && c1 == 1) || (f2 - f1 == 1 && c2 == 1)) {
            printf(YES);
        } else {
            printf(NO);
        }
    } else {
        printf(NO);
    }
    return 0;
}
