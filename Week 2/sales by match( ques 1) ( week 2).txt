#include <stdio.h>
int countSockPairs(int socks[], int n) {
    int pairs = 0;
    int visited[n];
    for(int i =0;i<n;i++){
        visited[i] = 0;
        
    }
    for(int i =0;i<n;i++){
        if(visited[i])
        continue;
        int count = 1;
        for(int j = i + 1; j < n; j++){
            if ( socks[i] == socks[j] && !visited[j]){
                count++;
                visited[j] = 1;
            }
        }
        pairs += count / 2;
    }
    return pairs;
}
int main(){
    int n;
    scanf("%d", &n);
    int socks[n];
    for(int i = 0; i < n; i++){
        scanf("%d", &socks[i]);
    }
    int result = countSockPairs(socks, n);
    printf("%d\n", result);
    return 0;
}
