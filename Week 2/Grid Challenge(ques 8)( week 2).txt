#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int compareChars(const void *a, const void *b) {
    return (*(char *)a - *(char *)b);
}
int isGridSorted(char **grid, int n) {
    for (int i = 0; i < n; i++) {
        qsort(grid[i], n, sizeof(char), compareChars);
    }
    for (int col = 0; col < n; col++) {
        for (int row = 0; row < n - 1; row++) {
            if (grid[row][col] > grid[row + 1][col]) {
                return 0; 
            }
        }
    }
     return 1; 
}
int main() {
    int t;
    scanf("%d", &t); 

    while (t--) {
        int n;
        scanf("%d", &n); 
        char **grid = malloc(n * sizeof(char *));
        for (int i = 0; i < n; i++) {
            grid[i] = malloc((n + 1) * sizeof(char)); 
            scanf("%s", grid[i]);
        }
        if (isGridSorted(grid, n)) {
            printf("YES\n");
        } else {
            printf("NO\n");
        }
        for (int i = 0; i < n; i++) {
            free(grid[i]);
        }
        free(grid);
    }

    return 0;
}
