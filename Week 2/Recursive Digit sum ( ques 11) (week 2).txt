#include <stdio.h>
#include <string.h>
int superDigit(long long n) {
    if (n < 10) return n;
    long long sum = 0;
    while (n > 0) {
        sum += n % 10;
        n /= 10;
    }
    return superDigit(sum);
}
int main() {
    char s[100001];
    int k;
    scanf("%s %d", s, &k);
    long long initialSum = 0;
    for (int i = 0; s[i] != '\0'; i++) {
        initialSum += (s[i] - '0');
    }
    long long total = initialSum * k;
    printf("%d\n", superDigit(total));

    return 0;
}
