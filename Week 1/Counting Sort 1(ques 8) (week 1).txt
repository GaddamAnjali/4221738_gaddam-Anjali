#include <stdio.h>
int main(){
    int n;
    scanf("%d", &n);
    int count[100] = {0};
    for(int i=0;i<n;i++){
        int value;
        scanf("%d", &value);
        if(value >= 0 && value < 100){
            count[value]++;
        }
    }
    for(int i=0;i<100;i++){
        printf("%d ", count[i]);
    }
    printf("\n");
    return 0;
}
