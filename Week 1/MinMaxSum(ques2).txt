#include <stdio.h>
void computeMinMaxSum(int numbers[]){
    long totalSum = 0;
    int smallest = numbers[0];
    int largest = numbers[0];
    for(int i=0;i<5;i++){
        int current = numbers[i];
        totalSum += current;
        if(current < smallest){
            smallest = current;
        }
        if(current>largest){
            largest=current;
        }
        
    }
    long minSum=totalSum-largest;
    long maxSum=totalSum-smallest;
    printf("%ld %ld\n",minSum, maxSum);
}
int main(){
    int input[5];
    for(int i=0;i<5;i++){
        scanf("%d", &input[i]);
    }
    computeMinMaxSum(input);
    return 0;
}
