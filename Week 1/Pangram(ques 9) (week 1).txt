#include <stdio.h>
#include <ctype.h>
int main(){
    char input[1001];
    int letters[26] = {0};
    int c;
    int index = 0;
    while ((c = getchar()) != '\n' && c != EOF && index < 1000){
        input[index++] = c;
    }
    input[index] = '\0';
    for(int i =0; input[i] != '\0'; i++){
        char ch = tolower(input[i]);
        if (ch >= 'a' && ch <= 'z'){
            letters[ch - 'a'] = 1;
        }
    }
    int isPangram = 1;
    for(int i=0;i<26;i++){
        if(letters[i]==0){
            isPangram = 0;
            break;
        }
    }
    if(isPangram){
        printf("pangram\n");
    }else {
        printf("not pangram\n");
    }
    return 0;
}
