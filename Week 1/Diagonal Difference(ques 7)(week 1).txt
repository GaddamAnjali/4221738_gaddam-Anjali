#include <stdio.h>
#include <stdlib.h>
int main(){
    int n;
    scanf("%d", &n);
    int matrix[n][n];
    for (int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            scanf("%d", &matrix[i][j]);
        }
    }
    int primarySum = 0;
    int secondarySum = 0;
    for (int i=0;i<n;i++){
        primarySum += matrix[i][i];
        secondarySum += matrix[i][n - 1 - i];
    }
    int diff = abs(primarySum - secondarySum);
    printf("%d\n", diff);
    return 0;
}
