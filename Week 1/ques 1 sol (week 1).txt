#include <stdio.h>

void findRatios(int arr[], int len) {
    int pos = 0, neg = 0, zero = 0;

    for (int i = 0; i < len; i++) {
        if (arr[i] > 0) pos++;
        else if (arr[i] < 0) neg++;
        else zero++;
    }

    printf("%.6f\n", (float)pos / len);
    printf("%.6f\n", (float)neg / len);
    printf("%.6f\n", (float)zero / len);
}

int main() {
    int n;
    scanf("%d", &n);
    int values[120]; 
    for (int j = 0; j < n; j++) {
        scanf("%d", &values[j]);
    }

    findRatios(values, n);
    return 0;
}
